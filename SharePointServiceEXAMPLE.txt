/**
 * SharePoint Service
 * Handles data persistence using either localStorage (mock mode) or SharePoint REST API (production)
 * 
 * Data Structure:
 * {
 *   lastUpdated: "ISO-8601 timestamp",
 *   units: [...],
 *   distributions: {...},
 *   giftTypes: [...]
 * }
 */

import { SHAREPOINT_CONFIG } from '../config/sharepoint.config';
import { INITIAL_UNITS, GIFT_TYPES } from '../data/initialUnits';

class SharePointService {
    constructor() {
        this.config = SHAREPOINT_CONFIG;
        this.useMock = this.config.useMock;
        console.log(`SharePointService initialized in ${this.useMock ? 'MOCK' : 'PRODUCTION'} mode`);
    }

    /**
     * Get the current data from storage (localStorage or SharePoint)
     * @returns {Promise<Object>} Data object with units, distributions, giftTypes
     */
    async getData() {
        if (this.useMock) {
            return this._getMockData();
        } else {
            return this._getSharePointData();
        }
    }

    /**
     * Update a distribution (checkbox toggle)
     * @param {string} unitName 
     * @param {string} giftType 
     * @param {Object} user - User object with displayName
     * @param {boolean} status - true to mark as distributed, false to unmark
     */
    async updateDistribution(unitName, giftType, user, status) {
        // Read-before-write for optimistic locking
        const data = await this.getData();

        const key = `${unitName}_${giftType}`;

        if (status) {
            // Mark as distributed
            data.distributions[key] = {
                status: true,
                markedBy: user.displayName || user.email || 'Unknown User',
                timestamp: new Date().toISOString()
            };
        } else {
            // Unmark (delete)
            delete data.distributions[key];
        }

        data.lastUpdated = new Date().toISOString();
        await this._saveData(data);

        return data;
    }

    /**
     * Update the units array
     * @param {Array} units 
     */
    async updateUnits(units) {
        const data = await this.getData();
        data.units = units;
        data.lastUpdated = new Date().toISOString();
        await this._saveData(data);
        return data;
    }

    /**
     * Update gift types array
     * @param {Array<string>} giftTypes 
     */
    async updateGiftTypes(giftTypes) {
        const data = await this.getData();
        data.giftTypes = giftTypes;
        data.lastUpdated = new Date().toISOString();
        await this._saveData(data);
        return data;
    }

    /**
     * Seed initial data if storage is empty
     */
    async seedInitialData() {
        try {
            const data = await this.getData();

            // Check if data is already seeded
            if (data.units && data.units.length > 0) {
                console.log('Data already seeded, skipping.');
                return false;
            }

            console.log('Seeding initial data...');

            // Convert INITIAL_UNITS to the correct format with IDs
            const units = INITIAL_UNITS.map((unit, index) => ({
                id: `unit_${index}`,
                name: unit.name,
                type: unit.type,
                order: index
            }));

            const initialData = {
                lastUpdated: new Date().toISOString(),
                units: units,
                distributions: {},
                giftTypes: GIFT_TYPES
            };

            await this._saveData(initialData);
            console.log('Initial data seeded successfully!');
            return true;
        } catch (error) {
            console.error('Error seeding initial data:', error);
            throw error;
        }
    }

    // ==========================================
    // MOCK MODE (localStorage) Implementation
    // ==========================================

    _getMockData() {
        try {
            const stored = localStorage.getItem(this.config.mockStorageKey);
            if (stored) {
                return Promise.resolve(JSON.parse(stored));
            }

            // Return empty structure if no data
            return Promise.resolve({
                lastUpdated: new Date().toISOString(),
                units: [],
                distributions: {},
                giftTypes: []
            });
        } catch (error) {
            console.error('Error reading mock data:', error);
            throw new Error('שגיאה בקריאת נתונים מהזיכרון המקומי');
        }
    }

    _saveMockData(data) {
        try {
            localStorage.setItem(this.config.mockStorageKey, JSON.stringify(data));
            return Promise.resolve(data);
        } catch (error) {
            console.error('Error saving mock data:', error);
            throw new Error('שגיאה בשמירת נתונים לזיכרון המקומי');
        }
    }

    // ==========================================
    // PRODUCTION MODE (SharePoint) Implementation
    // ==========================================

    async _getSharePointData() {
        try {
            const fileUrl = this.config.fileServerRelativeUrl;
            const endpoint = `/_api/web/GetFileByServerRelativeUrl('${fileUrl}')/$value`;

            const response = await fetch(endpoint, {
                method: 'GET',
                credentials: 'include', // Important for NTLM/Kerberos auth
                headers: {
                    'Accept': 'application/json;odata=verbose',
                }
            });

            if (!response.ok) {
                if (response.status === 404) {
                    // File doesn't exist yet, return empty structure
                    console.log('SharePoint file not found, returning empty data');
                    return {
                        lastUpdated: new Date().toISOString(),
                        units: [],
                        distributions: {},
                        giftTypes: []
                    };
                }
                throw new Error(`SharePoint request failed: ${response.status} ${response.statusText}`);
            }

            const text = await response.text();
            const data = JSON.parse(text);
            return data;
        } catch (error) {
            console.error('Error reading SharePoint data:', error);
            throw new Error('שגיאה בקריאת נתונים מ-SharePoint: ' + error.message);
        }
    }

    async _saveSharePointData(data) {
        try {
            // Step 1: Get security token (Form Digest)
            const digestResponse = await fetch('/_api/contextinfo', {
                method: 'POST',
                credentials: 'include',
                headers: {
                    'Accept': 'application/json;odata=verbose',
                }
            });

            if (!digestResponse.ok) {
                throw new Error('Failed to get security token from SharePoint');
            }

            const digestData = await digestResponse.json();
            const formDigestValue = digestData.d.GetContextWebInformation.FormDigestValue;

            // Step 2: Save the file
            const fileUrl = this.config.fileServerRelativeUrl;
            const endpoint = `/_api/web/GetFileByServerRelativeUrl('${fileUrl}')/$value`;

            const saveResponse = await fetch(endpoint, {
                method: 'POST',
                credentials: 'include',
                headers: {
                    'X-RequestDigest': formDigestValue,
                    'X-HTTP-Method': 'PUT', // Override POST to PUT
                    'IF-MATCH': '*', // Overwrite existing file
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify(data)
            });

            if (!saveResponse.ok) {
                throw new Error(`SharePoint save failed: ${saveResponse.status} ${saveResponse.statusText}`);
            }

            console.log('Data saved to SharePoint successfully');
            return data;
        } catch (error) {
            console.error('Error saving SharePoint data:', error);
            throw new Error('שגיאה בשמירת נתונים ל-SharePoint: ' + error.message);
        }
    }

    // ==========================================
    // Internal method dispatcher
    // ==========================================

    async _saveData(data) {
        if (this.useMock) {
            return this._saveMockData(data);
        } else {
            return this._saveSharePointData(data);
        }
    }
}

// Export singleton instance
export default new SharePointService();
